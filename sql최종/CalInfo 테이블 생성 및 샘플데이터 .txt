CREATE TABLE cal_Info(
    u_Idx NUMBER(5),
    cal_Idx NUMBER(1),
    cal_Category VARCHAR2(30), --영화 ,음식, 운동 
    cal_Thumbnail VARCHAR2(50) NOT NULL,
    cal_CreateAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY(u_Idx) REFERENCES customer(u_Idx) ON DELETE CASCADE,
    CONSTRAINT calendarinfo_PK PRIMARY KEY(cal_Idx, u_Idx)
);
